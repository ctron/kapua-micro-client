012package de.dentrassi.kapua.micro.client;
013
014import java.util.Collection;
015import java.util.HashMap;
016import java.util.Map;
017import java.util.Objects;
018
019import de.dentrassi.kapua.micro.client.lifecycle.BirthCertificateProvider;
020import de.dentrassi.kapua.micro.client.namespace.Namespace;
021import de.dentrassi.kapua.micro.client.transport.Transport;
022import de.dentrassi.kapua.micro.client.transport.TransportCreator;
023import de.dentrassi.kapua.micro.client.transport.TransportListener;
024import de.dentrassi.kapua.micro.client.util.Future;
025import de.dentrassi.kapua.micro.client.util.Nothing;
026
027/**
028 * A client instance for communicating with Kapua
029 */
030public class MicroClient implements AutoCloseable {
031
032    private final Namespace namespace;
033    private final BirthCertificateProvider[] birthCertificateProviders;
034    private final Transport transport;
035
036    private final Map<String, MicroApplication> applications = new HashMap<>();
037    private final TransportListener listener = new TransportListener() {
038
039        @Override
040        public void connected() {
041            handleConnnected();
042        }
043
044        @Override
045        public void disconnected() {
046            handleDisconnected();
047        }
048
049    };
050
051    public MicroClient(final Namespace namespace, final BirthCertificateProvider[] birthCertificateProviders, final TransportCreator<? extends Transport> transportCreator) throws Exception {
052        Objects.requireNonNull(namespace);
053        Objects.requireNonNull(transportCreator);
054
055        this.namespace = namespace;
056        this.birthCertificateProviders = birthCertificateProviders;
057        this.transport = transportCreator.createTransport(this.listener);
058    }
059
060    public synchronized MicroApplication createApplication(final String name) {
061        Objects.requireNonNull(name);
062
063        if (this.applications.containsKey(name)) {
064            throw new IllegalStateException("Application already exists:" + name);
065        }
066
067        final MicroApplication result = new MicroApplication(this, name);
068
069        this.applications.put(name, result);
070
071        sendBirthCertificate();
072
073        return result;
074    }
075
076    Future<Nothing> publish(final String applicationName, final Topic topic, final Payload payload) {
077        return this.transport.publish(this.namespace.data(applicationName, topic), payload);
078    }
079
080    Future<Nothing> subscribe(final String applicationName, final Topic topic, final Handler handler) {
081        return this.transport.subscribe(this.namespace.data(applicationName, topic), handler);
082    }
083
084    @Override
085    public void close() throws Exception {
086        if (this.transport != null) {
087            this.transport.close();
088        }
089    }
090
091    protected void handleConnnected() {
092        sendBirthCertificate();
093    }
094
095    protected void handleDisconnected() {
096    }
097
098    protected void sendBirthCertificate() {
099
100        // get topic
101
102        final String topic = this.namespace.birth();
103        if (topic == null) {
104            return;
105        }
106
107        final Payload.Builder payload = new Payload.Builder();
108
109        // gather payload data
110
111        if (this.birthCertificateProviders != null) {
112            for (final BirthCertificateProvider provider : this.birthCertificateProviders) {
113                if (provider == null) {
114                    continue;
115                }
116                provider.provide(payload);
117            }
118        }
119
120        // set the application IDs
121
122        {
123            final StringBuilder sb = new StringBuilder();
124            boolean first = true;
125            for (final String appId : this.applications.keySet()) {
126                if (!first) {
127                    sb.append(',');
128                } else {
129                    first = false;
130                }
131                sb.append(appId);
132            }
133
134            payload.metric("application_ids", sb.toString());
135        }
136
137        // send out
138
139        this.transport.publish(topic, payload.build());
140    }
141
142    synchronized void closeApplication(final String name, final Collection<Topic> topics) {
143
144        if (this.applications.remove(name) == null) {
145            return;
146        }
147
148        // unsubscribe from all subscriptions
149
150        if (!topics.isEmpty()) {
151            for (final Topic topic : topics) {
152                this.transport.unsubscribe(this.namespace.data(name, topic));
153            }
154        }
155    }
156
157}
