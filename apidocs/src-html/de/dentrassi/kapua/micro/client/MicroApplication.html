012package de.dentrassi.kapua.micro.client;
013
014import java.util.HashSet;
015import java.util.Set;
016
017import de.dentrassi.kapua.micro.client.util.Future;
018import de.dentrassi.kapua.micro.client.util.FutureTask;
019import de.dentrassi.kapua.micro.client.util.Nothing;
020
021/**
022 * An application instance
023 */
024public class MicroApplication implements AutoCloseable {
025
026    private final MicroClient client;
027    private final String name;
028    private final Set<Topic> subscriptions = new HashSet<>();
029
030    MicroApplication(final MicroClient client, final String name) {
031        this.client = client;
032        this.name = name;
033    }
034
035    public Future<Nothing> publish(final Topic topic, final Payload payload) {
036        return this.client.publish(this.name, topic, payload);
037    }
038
039    public Future<Nothing> subscribe(final Topic topic, final Handler handler) {
040        if (this.subscriptions.add(topic)) {
041            return this.client.subscribe(this.name, topic, handler);
042        } else {
043            return new FutureTask<Nothing>().completed(null);
044        }
045    }
046
047    @Override
048    public void close() throws Exception {
049        this.client.closeApplication(this.name, this.subscriptions);
050    }
051}
