012package de.dentrassi.kapua.micro.client.namespace;
013
014import java.util.Objects;
015
016import de.dentrassi.kapua.micro.client.Topic;
017import de.dentrassi.kapua.micro.client.transport.MqttTransportOptions;
018
019public class KuraNamespace implements Namespace {
020
021    private final String accountName;
022    private final String clientId;
023
024    public KuraNamespace(final String accountName, final MqttTransportOptions options) {
025        Objects.requireNonNull(accountName);
026        Objects.requireNonNull(options);
027        Objects.requireNonNull(options.getClientId());
028
029        this.accountName = accountName;
030        this.clientId = options.getClientId();
031    }
032
033    @Override
034    public String data(final String applicationName, final Topic topic) {
035        final StringBuilder sb = new StringBuilder();
036        sb.append(this.accountName).append('/');
037        sb.append(this.clientId).append('/');
038        sb.append(applicationName).append('/');
039        sb.append(topic);
040        return sb.toString();
041    }
042
043    @Override
044    public String birth() {
045        final StringBuilder sb = new StringBuilder("$EDC/");
046        sb.append(this.accountName).append('/');
047        sb.append(this.clientId).append("/MQTT/BIRTH");
048        return sb.toString();
049    }
050
051}
